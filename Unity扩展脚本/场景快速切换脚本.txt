using UnityEngine;
using UnityEditor;
using System.Collections.Generic;
using UnityEditor.SceneManagement;
using UnityEngine.SceneManagement;
using System.IO;

// 快速切换场景用
public class SceneSwitcherWindow : EditorWindow
{
    private List<string> sceneNames = new List<string>();
    private int selectedSceneIndex = 0;

    [MenuItem("Tools/Scene Switcher")]
    public static void ShowWindow()
    {
        GetWindow<SceneSwitcherWindow>("Scene Switcher");
    }

    private void OnEnable()
    {
        // 如果场景名称列表为空，则初始化
        if (sceneNames.Count == 0)
        {
            InitializeSceneNames();
        }
    }

    private void OnDisable()
    {
        // 清理场景名称列表
        sceneNames.Clear();
    }

    private void InitializeSceneNames()
    {
        // 获取所有已加载的场景名称
        for (int i = 0; i < SceneManager.sceneCountInBuildSettings; i++)
        {
            string scenePath = SceneUtility.GetScenePathByBuildIndex(i);
            string sceneName = Path.GetFileNameWithoutExtension(scenePath);
            sceneNames.Add(sceneName);            
        }
    }

    private void OnGUI()
    {
        // 创建一个下拉列表来选择场景
        if (sceneNames.Count > 0)
        {
            selectedSceneIndex = EditorGUILayout.Popup("选择场景", selectedSceneIndex, sceneNames.ToArray());

            // 创建一个按钮来切换场景
            if (GUILayout.Button("切换场景"))
            {
                // 保存当前场景的更改
                EditorSceneManager.SaveCurrentModifiedScenesIfUserWantsTo();

                // 切换到选择的场景
                Scene openedScene = EditorSceneManager.OpenScene(SceneUtility.GetScenePathByBuildIndex(selectedSceneIndex), OpenSceneMode.Single);                
            }
        }
        else
        {
            EditorGUILayout.LabelField("没有找到任何场景，请检查 Build Settings。");
        }
    }
}
